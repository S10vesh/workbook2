using System;
class Program
{
    static double Fact(int n)
    {
        double result = 1;
        for (int i = 2; i <= n; i++)
            result *= i;
        return result;
    }

    static double NNumb(double x, int n)
    {
        return Math.Pow(x, n) / Fact(n);
    }

    static double Calc(double x, double epsilon)
    {
        double sum = 0;
        double term;
        int n = 0;        
        do
        {
            term = Math.Pow(x, n) / Fact(n);
            sum += term;
            n++;
        }
        while (Math.Abs(term) > epsilon);
        
        return sum;
    }

    static void Main()
    {
        Console.WriteLine("Введите x:");
        double x = double.Parse(Console.ReadLine());
        
        Console.WriteLine("Введите точность (e < 0.01):");
        double epsilon = double.Parse(Console.ReadLine());
        if (epsilon >= 0.01)
        {
            Console.WriteLine("Точность должна быть меньше 0.01");
            return;
        }
        
        double result = Calc(x, epsilon);
        Console.WriteLine($"Значение функции с точностью {epsilon}: {result}");
        
        Console.WriteLine("\nВведите номер члена ряда (n):");
        int n = int.Parse(Console.ReadLine());
        double nthTerm = NNumb(x, n);
        Console.WriteLine($"Значение {n}-го члена ряда: {nthTerm}");
    }
}
